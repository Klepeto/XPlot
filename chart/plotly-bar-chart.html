open XPlot.Plotly

Plotly.Signin("Username", "API Key")

let data =
    Data(
        [
            Bar(
                x = ["giraffes"; "orangutans"; "monkeys"],
                y = [20; 14; 23]
            )
        ]
    )

let figure = Figure(data)

let plotlyResponse = figure.Plot("Basic Bar Chart")

figure.Show()
