let trace1 =
    Contour(
        z = z1,
        x = x1,
        y = y1,
        ncontours = 30.,
        showscale = false
    )

let trace2 =
    Scatter(
        x = x2,
        y = y2,
        mode = "markers+lines",
        name = "steepest",
        line = Line(color = "black")
    )

let multiData1 = Data([trace1; trace2])

let title = "A Contour and Scatter Plot of the Method of Steepest Descent"

let multiLayout1 = Layout(title = title)

Figure(multiData1, multiLayout1)
