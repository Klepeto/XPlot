        xbins =
            XBins(
                start = -3.2,

                line =
                    Line(
                        color = "grey",

                    ),
                opacity = 0.75

        x = coloredX1,
        name = "experimental",

                start = -1.8,

                size = 0.2

        marker = Marker(color = "rgb(255, 217, 102)"),
        opacity = 0.75
    )

let coloredLayout =
    Layout(
        title = "Colored and Styled Histograms",
        xaxis = XAxis(title = "Value"),
        yaxis = YAxis(title = "Count"),
        barmode = "overlay",

        bargroupgap = 0.3
